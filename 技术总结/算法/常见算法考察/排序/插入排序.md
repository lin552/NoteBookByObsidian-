/**  
 * 插入排序  
 */  
public class InsertionSort {  
  
    public static void main(String[] args) {  
        int[] originalArray = {5, 3, 8, 4, 6, 3, 9, 1, 7, 2};  
        int[] arr3 = Arrays.copyOf(originalArray, originalArray.length);  
        insertionSort(arr3);  
        printResult("插入排序", arr3);  
    }  
  
    /**  
     * 插入排序  
     * @param arr  
     */  
    public static void insertionSort(int[] arr) {  
        int n = arr.length;  
        for (int i = 1; i < n; i++) {  
            int key = arr[i];  
            int j = i - 1;  
            while (j >= 0 && arr[j] > key) {  
                arr[j + 1] = arr[j];  
                j--;  
            }  
            arr[j + 1] = key;  
        }  
    }  
  
    private static void printResult(String algorithm, int[] array) {  
        System.out.println(algorithm + ": " + Arrays.toString(array));  
    }  
}
